CSS Breakpoint Definitions
/* MaV Breakpoints — mobile first */ :root { --bp-xs: 360px; --bp-sm: 390px; --bp-md-sm:600px; --bp-md: 768px; --bp-lg: 1024px; --bp-xl: 1280px; --bp-2xl: 1440px; --bp-wide: 1920px; } /* Usage */ @media (min-width: 768px) { .card-grid { grid-template-columns: repeat(2, 1fr); } } @media (min-width: 1024px) { .card-grid { grid-template-columns: repeat(3, 1fr); } } /* Fluid type with clamp() */ .hero-title { font-size: clamp(28px, 5vw, 64px); }
